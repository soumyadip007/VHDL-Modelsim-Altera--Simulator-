library ieee;
use ieee.std_logic_1164.all;

entity half_adder is
    port
        ( a:in std_logic;
           b:in std_logic;
           diff:out std_logic;
           brw: out std_logic
        );
end half_adder;

architecture Behavioral of half_adder is 
begin 
    diff <= a xor b;
    brw<=  a and b;
end Behavioral;
